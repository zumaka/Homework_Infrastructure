FROM continuumio/miniconda3
LABEL "hw"="1"
LABEL "optimizatin"="true"
LABEL version="1"

SHELL ["/bin/bash", "-c"]
RUN conda create -n bio 

RUN echo "source activate bio" > ~/.bashrc
ENV PATH /opt/conda/envs/env/bin:$PATH

RUN apt-get update && apt-get install -y --no-install-recommends g++ gcc ca-certificates && rm -rf /var/lib/apt/lists/* \
 && conda install -y -c bioconda fastqc=0.11.8
RUN conda install -y -c bioconda STAR=2.7.10
RUN conda install -y -c bioconda samtools 
RUN conda install -y -c bioconda picard
RUN conda install -y -c bioconda salmon
RUN conda install -y -c bioconda  bedtools=2.30.0
